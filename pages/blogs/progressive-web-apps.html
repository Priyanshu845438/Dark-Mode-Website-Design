What are Progressive Web Apps?

Progressive Web Apps (PWAs) are web applications that use modern web capabilities to deliver an app-like experience to users. They combine the best of web and mobile apps, offering reliability, speed, and engagement that rivals native applications.

1. Core PWA Principles

PWA Characteristics:

  • Progressive: Works for every user, regardless of browser choice
  • Responsive: Fits any form factor: desktop, mobile, tablet
  • Connectivity Independent: Works offline or with poor connectivity
  • App-like: Feels like a native app with app-style interactions
  • Safe: Served via HTTPS to prevent tampering
  • Re-engageable: Makes re-engagement easy through push notifications
  • Installable: Can be installed on device home screen
  • Fresh: Always up-to-date thanks to service worker update process

3. Service Workers: The Heart of PWAs

Service workers enable offline functionality, background sync, and push notifications.

Basic Service Worker Setup

// sw.js - Service Worker
const CACHE_NAME = 'task-manager-v1';
const urlsToCache = [
  '/',
  '/styles/main.css',
  '/scripts/app.js',
  '/offline.html',
  '/icons/icon-192x192.png'
];

// Install event - cache resources
self.addEventListener('install', event => {
  event.waitUntil(
    caches.open(CACHE_NAME)
      .then(cache => {
        console.log('Opened cache');
        return cache.addAll(urlsToCache);
      })
  );
});

// Fetch event - serve from cache, fallback to network
self.addEventListener('fetch', event => {
  event.respondWith(
    caches.match(event.request)
      .then(response => {
        // Return cached resource if available
        if (response) {
          return response;
        }
        
        // Clone the request for fetch
        const fetchRequest = event.request.clone();
        
        return fetch(fetchRequest).then(response => {
          // Check if valid response
          if (!response || response.status !== 200 || response.type !== 'basic') {
            return response;
          }
          
          // Clone response for caching
          const responseToCache = response.clone();
          
          caches.open(CACHE_NAME)
            .then(cache => {
              cache.put(event.request, responseToCache);
            });
          
          return response;
        }).catch(() => {
          // Return offline page for navigation requests
          if (event.request.destination === 'document') {
            return caches.match('/offline.html');
          }
        });
      })
  );
});

Advanced Caching Strategies

// Advanced caching strategies
class CacheManager {
  constructor() {
    this.CACHE_NAMES = {
      static: 'static-cache-v1',
      dynamic: 'dynamic-cache-v1',
      api: 'api-cache-v1'
    };
  }
  
  // Cache First Strategy (for static assets)
  async cacheFirst(request) {
    const cache = await caches.open(this.CACHE_NAMES.static);
    const cachedResponse = await cache.match(request);
    
    if (cachedResponse) {
      return cachedResponse;
    }
    
    const networkResponse = await fetch(request);
    cache.put(request, networkResponse.clone());
    return networkResponse;
  }
  
  // Network First Strategy (for API calls)
  async networkFirst(request) {
    const cache = await caches.open(this.CACHE_NAMES.api);
    
    try {
      const networkResponse = await fetch(request);
      cache.put(request, networkResponse.clone());
      return networkResponse;
    } catch (error) {
      const cachedResponse = await cache.match(request);
      if (cachedResponse) {
        return cachedResponse;
      }
      throw error;
    }
  }
  
  // Stale While Revalidate Strategy (for dynamic content)
  async staleWhileRevalidate(request) {
    const cache = await caches.open(this.CACHE_NAMES.dynamic);
    const cachedResponse = await cache.match(request);
    
    const fetchPromise = fetch(request).then(networkResponse => {
      cache.put(request, networkResponse.clone());
      return networkResponse;
    });
    
    return cachedResponse || fetchPromise;
  }
}

4. Implementing Push Notifications

Push notifications help re-engage users and provide timely updates.

// Push notification implementation
class NotificationManager {
  async requestPermission() {
    const permission = await Notification.requestPermission();
    if (permission === 'granted') {
      console.log('Notification permission granted');
      return this.subscribeToNotifications();
    }
    return null;
  }
  
  async subscribeToNotifications() {
    const registration = await navigator.serviceWorker.ready;
    
    const subscription = await registration.pushManager.subscribe({
      userVisibleOnly: true,
      applicationServerKey: this.urlBase64ToUint8Array(this.vapidPublicKey)
    });
    
    // Send subscription to server
    await this.sendSubscriptionToServer(subscription);
    return subscription;
  }
  
  urlBase64ToUint8Array(base64String) {
    const padding = '='.repeat((4 - base64String.length % 4) % 4);
    const base64 = (base64String + padding)
      .replace(/-/g, '+')
      .replace(/_/g, '/');
    
    const rawData = window.atob(base64);
    const outputArray = new Uint8Array(rawData.length);
    
    for (let i = 0; i < rawData.length; ++i) {
      outputArray[i] = rawData.charCodeAt(i);
    }
    return outputArray;
  }
  
  async sendSubscriptionToServer(subscription) {
    return fetch('/api/notifications/subscribe', {
      method: 'POST',
      headers: {
        'Content-Type': 'application/json',
      },
      body: JSON.stringify(subscription)
    });
  }
}

5. Background Sync

Background sync allows your app to defer actions until the user has stable connectivity.

// Background sync implementation
class BackgroundSync {
  async registerSync(tag) {
    const registration = await navigator.serviceWorker.ready;
    return registration.sync.register(tag);
  }
  
  async syncTasks() {
    // Get pending tasks from IndexedDB
    const pendingTasks = await this.getPendingTasks();
    
    if (pendingTasks.length === 0) {
      return;
    }
    
    try {
      // Sync with server
      for (const task of pendingTasks) {
        await fetch('/api/tasks', {
          method: 'POST',
          body: JSON.stringify(task),
          headers: {
            'Content-Type': 'application/json'
          }
        });
        
        // Remove from pending queue
        await this.removePendingTask(task.id);
      }
    } catch (error) {
      console.error('Sync failed:', error);
      // Tasks remain in queue for next sync attempt
    }
  }
}

// In service worker - handle background sync
self.addEventListener('sync', event => {
  if (event.tag === 'task-sync') {
    event.waitUntil(
      syncTasks()
    );
  }
});

async function syncTasks() {
  const db = await openDB('TaskManager', 1);
  const tx = db.transaction('pendingTasks', 'readonly');
  const store = tx.objectStore('pendingTasks');
  const pendingTasks = await store.getAll();
  
  for (const task of pendingTasks) {
    try {
      const response = await fetch('/api/tasks', {
        method: 'POST',
        body: JSON.stringify(task.data),
        headers: {
          'Content-Type': 'application/json'
        }
      });
      
      if (response.ok) {
        // Remove from pending queue
        const deleteTx = db.transaction('pendingTasks', 'readwrite');
        const deleteStore = deleteTx.objectStore('pendingTasks');
        await deleteStore.delete(task.id);
      }
    } catch (error) {
      console.error('Failed to sync task:', error);
    }
  }
}

6. App Shell Architecture

The app shell model separates the core application infrastructure from the data.

// App shell implementation
class AppShell {
  constructor() {
    this.shell = {
      header: document.querySelector('#app-header'),
      navigation: document.querySelector('#app-nav'),
      content: document.querySelector('#app-content'),
      footer: document.querySelector('#app-footer')
    };
    
    this.loadShell();
  }
  
  async loadShell() {
    // Load cached shell components
    const cacheShell = await this.getCachedShell();
    if (cacheShell) {
      this.renderShell(cacheShell);
    }
    
    // Load content dynamically
    this.loadContent();
  }
  
  async getCachedShell() {
    const cache = await caches.open('app-shell-v1');
    const shellResponse = await cache.match('/shell.html');
    
    if (shellResponse) {
      return await shellResponse.text();
    }
    
    return null;
  }
  
  renderShell(shellHTML) {
    const parser = new DOMParser();
    const shellDoc = parser.parseFromString(shellHTML, 'text/html');
    
    // Update shell components
    this.shell.header.innerHTML = shellDoc.querySelector('#app-header').innerHTML;
    this.shell.navigation.innerHTML = shellDoc.querySelector('#app-nav').innerHTML;
    this.shell.footer.innerHTML = shellDoc.querySelector('#app-footer').innerHTML;
  }
  
  async loadContent() {
    const path = window.location.pathname;
    
    try {
      // Try to load from cache first
      const cache = await caches.open('dynamic-content-v1');
      let response = await cache.match(path);
      
      if (!response) {
        // Fetch from network
        response = await fetch(`/api/content${path}`);
        
        if (response.ok) {
          cache.put(path, response.clone());
        }
      }
      
      const content = await response.json();
      this.renderContent(content);
    } catch (error) {
      this.renderOfflineContent();
    }
  }
  
  renderContent(content) {
    this.shell.content.innerHTML = content.html;
    
    // Update page title and meta tags
    document.title = content.title;
    
    // Trigger content-specific JavaScript
    this.initializeContentScripts();
  }
  
  renderOfflineContent() {
    this.shell.content.innerHTML = `
      <div class="offline-message">
        <h2>You're offline</h2>
        <p>Please check your connection and try again.</p>
        <button onclick="window.location.reload()">Retry</button>
      </div>
    `;
  }
}

7. Performance Optimization

PWA Performance Best Practices:

  • Critical Resource Prioritization: Load critical CSS and JS first
  • Code Splitting: Split code into smaller, cacheable chunks
  • Lazy Loading: Load resources only when needed
  • Service Worker Optimization: Efficient caching strategies
  • Bundle Optimization: Tree shaking and minification
